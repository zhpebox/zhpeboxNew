public class Jha{

	public static void main(String args[]){
		int[] a = new int[args.length];
		for(int i = 0;i<a.length;i++){
			a[i] = Integer.parseInt(args[i]); 
		}
		
		for(int i = 0 ;i<args.length;i++){
			for(int j = i+1;j<args.length;j++){
				if(a[i] < a[j]){
					int m = a[i];
					a[i] = a[j];
					a[j] = m;
				}
			}
		}
		for(int e : a){
			System.out.println(e);
		}
	} 
}
